apprunner.yaml

version: 1.0
runtime: nodejs22
build:
  commands:
    pre-build:
      - npm install
    build:
      - npm run build
run:
  command: npm start
  network:
    port: 8080
  env:
    - name: NODE_ENV
      value: production

AWS CLI Commands

# Create GitHub connection
aws apprunner create-connection \
  --connection-name github-app-runner

# Create App Runner service
aws apprunner create-service \
  --cli-input-json file://config.json

# Get live URL
aws apprunner list-services \
  --query "ServiceSummaryList[].ServiceUrl"
